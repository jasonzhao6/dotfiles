Shortcuts: main_keymap.github_desktop.zsh

  cmd-1       # Show changes
  cmd-2       # Show history
  cmd-t       # Show repositories list
  cmd-b       # Show branches list
  cmd-g       # Go to summary

  cmd-p       # Push
  cmd-P       # Pull

  cmd-shift-g # View on GitHub
  cmd-shift-f # Show in Finder
  cmd-shift-a # Open in TextMate
  ctrl-`      # Open in Terminal
