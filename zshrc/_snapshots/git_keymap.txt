Keymap: git_keymap.zsh

  `   1   2   3   4   5   |   6   7   8   9   0   [   ] 
      '   ,   .  <p> (y)  |  (f) (g) <c> (r) (l)  /   =   \ 
     <a> <o> <e> <u> (i)  |  <d> <h> <t> <n> <s>  - 
      ;   q   j  <k> (x)  |  (b) <m> <w> (v) (z)

  `<>` key initials have one mapping
  `()` key initials have multiple mappings

Keymap Usage

  $ g              # Show this keymap
  $ g <regex>      # Search this keymap

  $ g.<key>        # This mapping takes no variable
  $ g.<key> <var>  # This mapping takes one variable
  $ g.<key> <var>? # This mapping takes zero or one variable
  $ g.<key> <var>* # This mapping takes zero or multiple variables
  $ g.<key> (1-10) # This mapping takes an exact value from the list

     ^             # The `.` is only for documentation
                   # Omit it when invoking a mapping

Keymap List

  $ g.g            # Checkout latest `main`
  $ g.n <branch>   # Checkout a new branch
  $ g.k <branch>   # Checkout a remote branch

  $ g.b            # List branches
  $ g.bb           # Delete merged branches
  $ g.bd <branch>  # Delete specified branch
  $ g <branch>     # Checkout a local branch

  $ g.d            # Diff
  $ g.t            # Status
  $ g.c            # Create new commit
  $ g.m            # Amend previous commit
  $ g.w            # Reword previous commit
  $ g.i <sha>?     # Fixup commit (Default: Pasteboard)
  $ g.ii           # List last 20 commits
  $ g.e            # Create empty commit

  $ g.x <number>?  # Rebase on main (or last N commits)
  $ g.xm           # Rebase on master
  $ g.xu           # Rebase on upstream
  $ g.xc           # Rebase continue
  $ g.xa           # Rebase abort

  $ g.y <sha>?     # Cherry pick (Default: Pasteboard)
  $ g.yc           # Cherry pick continue
  $ g.ya           # Cherry pick abort

  $ g.v <sha>?     # Revert commit (Default: Pasteboard)
  $ g.vc           # Revert continue
  $ g.va           # Revert abort

  $ g.u <number>?  # Undo last N commits (Default: 1)
  $ g.z <number>?  # Discard changes (& last N commits)
  $ g.zz           # Discard changes & last commit

  $ g.s <message>? # Stash
  $ g.a <index>?   # Apply a stash (Default: Latest)
  $ g.l            # List stashes
  $ g.lc           # Clear stashes

  $ g.r <match>*   # List logs & filter
  $ g.rr <match>*  # List first-parent logs & filter

  $ g.P            # Pull
  $ g.p            # Push
  $ g.f            # Force push with lease
  $ g.ff           # Force push

  $ g.h            # (Reserved: GitHub CLI)
  $ g.o            # (Reserved: Go language)
